def jouer(unX, unY, rougeOuBleu)
      case rougeOuBleu
              when 'b', 'bleu'
                      @matrice[unX][unY].setBleu
              when 'r', 'rouge'
                      @matrice[unX][unY].setRouge
              when 'v', 'vide'
                      @matrice[unX][unY].setVide
      end
end
