def resetTouche(unPseudo)
        case @id
                when 1
                        BaseDeDonnees.setRaccourci(unPseudo, @id,'p')
                when 2
                        BaseDeDonnees.setRaccourci(unPseudo, @id,'d')
                when 3
                        BaseDeDonnees.setRaccourci(unPseudo, @id,'f')
                when 4
                        BaseDeDonnees.setRaccourci(unPseudo, @id,'h')
        end
end
