def on_buttonDix_toggled
        if(@tailleChoisie != nil)
                case @tailleChoisie
                        when 6
                                @buttonSix.set_active(false)
                        when 8
                                @buttonHuit.set_active(false)
                        when 12
                                @buttonDouze.set_active(false)
                end
        end
        @tailleChoisie = 10
end
